        \begin{equation*}
            \breve{P}_e = \frac{1}{2} \left(1-\norm{p_1 \pmb{\Xi}_1 - p_0 \pmb{\Xi}_0}_1 \right)

    \begin{block}<2->{MDEP for pure states}

            \breve{P}_e = \frac{1}{2} \left(1- \sqrt{1-4 p_0 p_1 \absolutevalue{

        \end{equation*}
    \end{block}
